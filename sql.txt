-- ==========================================
-- 0. CLEAN START (DROP EVERYTHING)
-- ==========================================

DROP TABLE IF EXISTS favoritos, mensajes, conversaciones, eventos_envio, incidencias, costes_pedidos, pedidos, ofertas, fotos_articulo, articulos, usuarios CASCADE;
DROP TYPE IF EXISTS estado_usuario_enum, estado_articulo_enum, estado_oferta_enum, estado_pedido_enum, tipo_coste_enum, tipo_incidencia_enum, estado_incidencia_enum CASCADE;

-- ==========================================
-- 1. TYPES & ENUMS
-- ==========================================

CREATE TYPE estado_usuario_enum AS ENUM ('activo', 'inactivo', 'suspendido', 'eliminado');
CREATE TYPE estado_articulo_enum AS ENUM ('disponible', 'reservado', 'vendido', 'desactivado', 'eliminado');
CREATE TYPE estado_oferta_enum AS ENUM ('pendiente', 'aceptada', 'rechazada', 'caducada');
CREATE TYPE estado_pedido_enum AS ENUM ('pendiente_pago', 'pagado', 'enviado', 'en_reparto', 'entregado', 'completado', 'cancelado');
CREATE TYPE tipo_coste_enum AS ENUM ('envio', 'comision_comprador', 'seguro_proteccion', 'extra_vendedor');
CREATE TYPE tipo_incidencia_enum AS ENUM ('no_recibido', 'articulo_diferente', 'articulo_danado', 'falsificacion');
CREATE TYPE estado_incidencia_enum AS ENUM ('abierta', 'en_mediacion', 'resuelta_reembolso', 'resuelta_sin_reembolso', 'cerrada');

-- ==========================================
-- 2. TABLES
-- ==========================================

CREATE TABLE usuarios (
    id_usuario UUID PRIMARY KEY, -- Sync with Supabase Auth
    email VARCHAR(255) NOT NULL UNIQUE,
    estado estado_usuario_enum DEFAULT 'activo' NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE articulos (
    id_articulo BIGSERIAL PRIMARY KEY,
    id_vendedor UUID NOT NULL,
    titulo VARCHAR(150) NOT NULL,
    descripcion TEXT,
    categoria VARCHAR(100) NOT NULL,
    precio_base NUMERIC(10, 2) NOT NULL,
    estado_articulo estado_articulo_enum DEFAULT 'disponible' NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_vendedor) REFERENCES usuarios(id_usuario) ON DELETE CASCADE
);

CREATE TABLE fotos_articulo (
    id_foto BIGSERIAL PRIMARY KEY,
    id_articulo BIGINT NOT NULL,
    image_url TEXT NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_articulo) REFERENCES articulos(id_articulo) ON DELETE CASCADE
);

CREATE TABLE ofertas (
    id_oferta BIGSERIAL PRIMARY KEY,
    id_comprador UUID NOT NULL,
    id_articulo BIGINT NOT NULL,
    ultimo_emisor_id UUID REFERENCES usuarios(id_usuario), -- Quién hizo la última propuesta
    estado estado_oferta_enum DEFAULT 'pendiente' NOT NULL,
    importe NUMERIC(10, 2) NOT NULL,
    mensaje TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_comprador) REFERENCES usuarios(id_usuario) ON DELETE CASCADE,
    FOREIGN KEY (id_articulo) REFERENCES articulos(id_articulo) ON DELETE CASCADE
);

CREATE TABLE pedidos (
    id_pedido BIGSERIAL PRIMARY KEY,
    id_comprador UUID NOT NULL,
    id_articulo BIGINT NOT NULL,
    estado_pedido estado_pedido_enum DEFAULT 'pendiente_pago' NOT NULL,
    precio_final NUMERIC(10, 2) NOT NULL,
    metodo_envio VARCHAR(100),
    desde_direccion TEXT NOT NULL,
    hacia_direccion TEXT NOT NULL,
    tracking_number VARCHAR(150),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_comprador) REFERENCES usuarios(id_usuario) ON DELETE RESTRICT,
    FOREIGN KEY (id_articulo) REFERENCES articulos(id_articulo) ON DELETE RESTRICT
);

CREATE TABLE costes_pedidos (
    id_coste BIGSERIAL PRIMARY KEY,
    id_pedido BIGINT NOT NULL,
    tipo_coste tipo_coste_enum NOT NULL,
    importe NUMERIC(10, 2) NOT NULL,
    moneda VARCHAR(3) DEFAULT 'EUR' NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_pedido) REFERENCES pedidos(id_pedido) ON DELETE CASCADE
);

CREATE TABLE incidencias (
    id_incidencia BIGSERIAL PRIMARY KEY,
    id_pedido BIGINT NOT NULL,
    tipo_incidencia tipo_incidencia_enum NOT NULL,
    motivo TEXT NOT NULL,
    estado_incidencia estado_incidencia_enum DEFAULT 'abierta' NOT NULL,
    importe_reembolso NUMERIC(10, 2),
    requiere_devolucion BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_pedido) REFERENCES pedidos(id_pedido) ON DELETE CASCADE
);

CREATE TABLE eventos_envio (
    id_evento BIGSERIAL PRIMARY KEY,
    id_pedido BIGINT NOT NULL,
    descripcion_evento TEXT NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_pedido) REFERENCES pedidos(id_pedido) ON DELETE CASCADE
);

CREATE TABLE conversaciones (
    id_conversacion BIGSERIAL PRIMARY KEY,
    id_usuario_1 UUID NOT NULL,
    id_usuario_2 UUID NOT NULL,
    id_articulo BIGINT NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_usuario_1) REFERENCES usuarios(id_usuario) ON DELETE CASCADE,
    FOREIGN KEY (id_usuario_2) REFERENCES usuarios(id_usuario) ON DELETE CASCADE,
    FOREIGN KEY (id_articulo) REFERENCES articulos(id_articulo) ON DELETE CASCADE,
    CONSTRAINT unique_conversacion UNIQUE (id_usuario_1, id_usuario_2, id_articulo)
);

CREATE TABLE mensajes (
    id_mensaje BIGSERIAL PRIMARY KEY,
    id_conversacion BIGINT NOT NULL,
    id_emisor UUID NOT NULL,
    contenido TEXT NOT NULL,
    leido BOOLEAN DEFAULT FALSE NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_conversacion) REFERENCES conversaciones(id_conversacion) ON DELETE CASCADE,
    FOREIGN KEY (id_emisor) REFERENCES usuarios(id_usuario) ON DELETE CASCADE
);

CREATE TABLE favoritos (
    id_favorito BIGSERIAL PRIMARY KEY,
    id_usuario UUID NOT NULL,
    id_articulo BIGINT NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id_usuario) ON DELETE CASCADE,
    FOREIGN KEY (id_articulo) REFERENCES articulos(id_articulo) ON DELETE CASCADE,
    CONSTRAINT unique_favorito UNIQUE (id_usuario, id_articulo)
);

-- ==========================================
-- 3. RLS POLICIES
-- ==========================================

-- Usuarios
ALTER TABLE usuarios ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Usuarios pueden ver su propio perfil" ON usuarios FOR SELECT USING (auth.uid() = id_usuario);
CREATE POLICY "Sistema puede insertar usuarios" ON usuarios FOR INSERT WITH CHECK (true);

-- Artículos
ALTER TABLE articulos ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Cualquiera puede ver artículos disponibles" ON articulos FOR SELECT USING (estado_articulo = 'disponible' OR auth.uid() = id_vendedor);
CREATE POLICY "Usuarios pueden crear artículos" ON articulos FOR INSERT WITH CHECK (auth.uid() = id_vendedor);
CREATE POLICY "Vendedores pueden editar sus artículos" ON articulos FOR UPDATE USING (auth.uid() = id_vendedor);

-- Fotos
ALTER TABLE fotos_articulo ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Ver fotos de artículos" ON fotos_articulo FOR SELECT USING (true);
CREATE POLICY "Sellers pueden subir fotos" ON fotos_articulo FOR INSERT WITH CHECK (
    EXISTS (SELECT 1 FROM articulos WHERE id_articulo = fotos_articulo.id_articulo AND id_vendedor = auth.uid())
);

-- Ofertas
ALTER TABLE ofertas ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Participantes pueden ver ofertas" ON ofertas FOR SELECT USING (
    auth.uid() = id_comprador OR EXISTS (SELECT 1 FROM articulos WHERE id_articulo = ofertas.id_articulo AND id_vendedor = auth.uid())
);
CREATE POLICY "Compradores pueden crear ofertas" ON ofertas FOR INSERT WITH CHECK (auth.uid() = id_comprador);
CREATE POLICY "Participantes pueden actualizar ofertas" ON ofertas FOR UPDATE USING (
    auth.uid() = id_comprador OR EXISTS (SELECT 1 FROM articulos WHERE id_articulo = ofertas.id_articulo AND id_vendedor = auth.uid())
);

-- Conversaciones
ALTER TABLE conversaciones ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Usuarios pueden ver sus chats" ON conversaciones FOR SELECT USING (auth.uid() = id_usuario_1 OR auth.uid() = id_usuario_2);
CREATE POLICY "Usuarios pueden crear chats" ON conversaciones FOR INSERT WITH CHECK (auth.uid() = id_usuario_1 OR auth.uid() = id_usuario_2);

-- Mensajes
ALTER TABLE mensajes ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Usuarios pueden ver mensajes de sus chats" ON mensajes FOR SELECT USING (
  EXISTS (SELECT 1 FROM conversaciones WHERE id_conversacion = mensajes.id_conversacion AND (id_usuario_1 = auth.uid() OR id_usuario_2 = auth.uid()))
);
CREATE POLICY "Usuarios pueden enviar mensajes" ON mensajes FOR INSERT WITH CHECK (auth.uid() = id_emisor);

-- Favoritos
ALTER TABLE favoritos ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Usuarios ven sus propios favoritos" ON favoritos FOR SELECT USING (auth.uid() = id_usuario);
CREATE POLICY "Usuarios pueden añadir favoritos" ON favoritos FOR INSERT WITH CHECK (auth.uid() = id_usuario);
CREATE POLICY "Usuarios pueden quitar favoritos" ON favoritos FOR DELETE USING (auth.uid() = id_usuario);
